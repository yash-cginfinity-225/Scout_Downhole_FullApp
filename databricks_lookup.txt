LOOKUP_TABLE = "databricksnonprod.default.lookup_table_config"

TABLE_MAP = {
    "bha_tally":         "databricksnonprod.default.halliburton_bha_tally",
    "bha_report":        "databricksnonprod.default.halliburton_bha_report",
    "bha_extracted":     "databricksnonprod.default.bha_extracted_reports",
    "motor_performance": "databricksnonprod.default.halliburton_motor_performance",
}

# Order must match how mapped_columns are stored (bha_tally,bha_report,bha_extracted,motor_performance)
TABLE_KEYS = ["bha_tally", "bha_report", "bha_extracted", "motor_performance"]

# ── 1. Read lookup config ─────────────────────────────────────────────────────
lookup_rows = spark.table(LOOKUP_TABLE).collect()

mappings = []
for row in lookup_rows:
    mapped_cols = (row["mapped_columns"] or "").split(",")
    while len(mapped_cols) < 4:
        mapped_cols.append("N/A")
    sub_fields_str = row["sub_fields"] or ""
    sub_fields = [s.strip() for s in sub_fields_str.split(",") if s.strip()]
    mappings.append({
        "column_name": row["column_name"],
        "mapped_columns": [c.strip() for c in mapped_cols],
        "sub_fields": sub_fields,
    })

# ── 2. Determine output columns (same order as admin page) ────────────────────
output_columns = ["file_name", "source_table"]
for m in mappings:
    if m["sub_fields"]:
        for sf in m["sub_fields"]:
            col_key = f"{m['column_name']}__{sf}"
            if col_key not in output_columns:
                output_columns.append(col_key)
    else:
        if m["column_name"] not in output_columns:
            output_columns.append(m["column_name"])

# ── 3. Query each source table and build unified wide rows ────────────────────
all_rows = []

for table_idx, table_key in enumerate(TABLE_KEYS):
    full_table = TABLE_MAP[table_key]

    # Which source columns are needed from this table
    col_mapping = {}    # source_col -> display_name
    sub_field_map = {}  # source_col -> [sub_field, ...]

    for m in mappings:
        source_col = m["mapped_columns"][table_idx].strip()
        if source_col and source_col != "N/A":
            col_mapping[source_col] = m["column_name"]
            if m["sub_fields"]:
                sub_field_map[source_col] = m["sub_fields"]

    if not col_mapping:
        continue  # This table has no mappings

    df = spark.table(full_table)
    avail = set(df.columns)

    # Select path + only the columns that exist in the table
    select_cols = ["path"] + [c for c in col_mapping if c in avail]
    df_rows = df.select(*select_cols).collect()

    for row in df_rows:
        path_val = row["path"] or ""
        file_name = path_val.rsplit("/", 1)[-1] if path_val else ""

        # Start with a blank wide row
        unified = {c: None for c in output_columns}
        unified["file_name"] = file_name
        unified["source_table"] = table_key

        for source_col, display_name in col_mapping.items():
            if source_col not in row.asDict():
                continue

            value = row[source_col]

            # Parse JSON array strings into Python lists
            if isinstance(value, str) and value.strip().startswith("["):
                try:
                    value = json.loads(value)
                except Exception:
                    pass

            if source_col in sub_field_map and isinstance(value, list):
                # Expand nested array → one column per sub-field
                for sf in sub_field_map[source_col]:
                    sf_values = [
                        str(item[sf])
                        for item in value
                        if isinstance(item, dict) and sf in item and item[sf] is not None
                    ]
                    unified[f"{display_name}__{sf}"] = ", ".join(sf_values) if sf_values else None
            elif isinstance(value, list):
                unified[display_name] = ", ".join(str(v) for v in value if v is not None)
            else:
                unified[display_name] = value

        all_rows.append(unified)

# ── 4. Build final DataFrame and display ──────────────────────────────────────
if all_rows:
    result_df = spark.createDataFrame(all_rows)
    result_df = result_df.select(*output_columns)
    display(result_df)
else:
    print("No mapped data found. Check that lookup_table_config has entries and source tables are populated.")
